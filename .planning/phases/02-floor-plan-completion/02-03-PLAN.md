Add a new exported function `getReconnectionContext` that determines what space the user should join:
```typescript
export interface ReconnectionContext {
  type: 'grace-rejoin' | 'home-space' | 'default-space' | 'fallback' | 'first-time';
  spaceId: string | null;
  spaceName?: string;
  reason: string;
}

export function getReconnectionContext(
  currentUser: User,
  spaces: Space[],
  company: Company | null,
  lastSpaceId: string | null,
): ReconnectionContext {
  // Step 1: Check grace period
  const disconnectTsStr = typeof window !== 'undefined'
    ? localStorage.getItem(DISCONNECT_TS_KEY)
    : null;
  const disconnectTs = disconnectTsStr ? parseInt(disconnectTsStr, 10) : 0;
  const withinGrace = disconnectTs > 0 && (Date.now() - disconnectTs) < GRACE_PERIOD_MS;

  if (withinGrace && lastSpaceId) {
    const lastSpace = spaces.find(s => s.id === lastSpaceId && s.status === 'active');
    if (lastSpace) {
      return {
        type: 'grace-rejoin',
        spaceId: lastSpace.id,
        spaceName: lastSpace.name,
        reason: 'Within 5-minute grace period',
      };
    }
    // Last space deleted/inactive -- fall through to home/default
  }

  // Step 2: Check if first-time user
  const isFirstTime = typeof window !== 'undefined'
    ? !localStorage.getItem(FIRST_LOGIN_KEY)
    : true;

  if (isFirstTime) {
    const defaultSpaceId = company?.settings?.defaultSpaceId;
    if (defaultSpaceId) {
      const defaultSpace = spaces.find(s => s.id === defaultSpaceId && s.status === 'active');
      if (defaultSpace) {
        return {
          type: 'first-time',
          spaceId: defaultSpace.id,
          spaceName: defaultSpace.name,
          reason: 'First login -- placed in company default space',
        };
      }
    }
    // No default space set -- fallback
    const firstWorkspace = spaces.find(s => s.type === 'workspace' && s.status === 'active');
    if (firstWorkspace) {
      return {
        type: 'first-time',
        spaceId: firstWorkspace.id,
        spaceName: firstWorkspace.name,
        reason: 'First login -- no default space set, using first workspace',
      };
    }
    return { type: 'fallback', spaceId: null, reason: 'No active spaces available' };
  }

  // Step 3: Returning user -- use home space
  const homeSpaceId = company?.settings?.homeSpaces?.[currentUser.id];
  if (homeSpaceId) {
    const homeSpace = spaces.find(s => s.id === homeSpaceId && s.status === 'active');
    if (homeSpace) {
      return {
        type: 'home-space',
        spaceId: homeSpace.id,
        spaceName: homeSpace.name,
        reason: 'Returning user -- assigned home space',
      };
    }
    // Home space deleted/maintenance -- fall through
  }

  // Step 4: No home space -- use company default
  const defaultSpaceId = company?.settings?.defaultSpaceId;
  if (defaultSpaceId) {
    const defaultSpace = spaces.find(s => s.id === defaultSpaceId && s.status === 'active');
    if (defaultSpace) {
      return {
        type: 'default-space',
        spaceId: defaultSpace.id,
        spaceName: defaultSpace.name,
        reason: 'No home space assigned -- using company default',
      };
    }
  }

  // Step 5: Ultimate fallback -- first active workspace-type space
  const firstWorkspace = spaces.find(s => s.type === 'workspace' && s.status === 'active');
  if (firstWorkspace) {
    return {
      type: 'fallback',
      spaceId: firstWorkspace.id,
      spaceName: firstWorkspace.name,
      reason: 'No home or default space -- first active workspace',
    };
  }

  return { type: 'fallback', spaceId: null, reason: 'No active spaces available' };
}
```

const updateCompanyDetails = async (data: Partial<Company>) => {
  if (!company) return;
  try {
    // Deep merge settings to preserve existing fields
    const mergedData = { ...data };
    if (data.settings && company.settings) {
      mergedData.settings = { ...company.settings, ...data.settings };
    }
    await updateCompany(company.id, mergedData);
    setCompany(prev => {
      if (!prev) return null;
      const updated = { ...prev, ...mergedData };
      if (mergedData.settings) {
        updated.settings = { ...prev.settings, ...mergedData.settings };
      }
      return updated;
    });
  } catch (error) {
    throw error;
  }
};
