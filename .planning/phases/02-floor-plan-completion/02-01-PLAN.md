```typescript
const handleBannerApprove = useCallback(async (request: KnockRequestPayload) => {
  // Clear the banner immediately
  setPendingKnockRequests(prev => {

const handleBannerDeny = useCallback(async (request: KnockRequestPayload) => {
```

**4. Add knock initiation handler:**

```typescript
const handleKnockOnSpace = useCallback(async (spaceId: string) => {
  if (!currentUserProfile || !knock.canKnock(spaceId)) return;
  knock.knock(spaceId);
  try {
    const res = await fetch('/api/spaces/knock/request', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ spaceId }),
    });
    if (!res.ok) {
      const data = await res.json().catch(() => ({}));
      toast.error('Could not send knock request. Check your connection and try again.');
      knock.cancel();
      return;
    }
    const data = await res.json();
    // API always returns `requestId` (verified: src/app/api/spaces/knock/request/route.ts line 108)
    activeKnockRequestIdRef.current = data.requestId;
  } catch {
    toast.error('Could not send knock request. Check your connection and try again.');
    knock.cancel();
  }
}, [currentUserProfile, knock]);
```
